## 代码

```
/* 先序遍历 */
void PreOrderTraversal(BinTree BT) {
    if (BT) {
        printf(" %c", BT->Data);
        PreOrderTraversal(BT->Left);
        PreOrderTraversal(BT->Right);   //递归用法
    }
}

/* 中序遍历 */
void InOrderTravelsal(BinTree BT) {
    if (BT) {
        InOrderTraversal(BT->Left);
        printf(" %c", BT->Data);
        InOrderTraversal(BT->Right);   //递归用法
    }
}

/* 后序遍历 */
void PostOrderTravelsal(BinTree BT) {
    if (BT) {
        PostOrderTraversal(BT->Left);
        PostOrderTraversal(BT->Right);   //递归用法
        printf(" %c", BT->Data);
    }
}

/* 层序遍历 */
void LevelorderTraversal( BinTree BT ) {
    BinTree t[101];
    int i, j;
    i = j = 0;
    t[0] = BT;
    while(BT) {
        if (BT->Left) {
            t[++i] = BT->Left;
        }
        if (BT->Right) {
            t[++i] = BT->Right;
        }
        
        BinTree temp = t[j];
        printf(" %c", temp->Data);
        if (i == j) {
            break;
        }
        j++;
        BT = t[j];
    }
}
```
